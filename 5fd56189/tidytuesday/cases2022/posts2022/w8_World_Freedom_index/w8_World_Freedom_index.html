library(sf)
world1 <- ne_countries(scale=110,type="countries",

world12 <- world1%>%
  as.data.frame()%>%
  select(country=sovereignt,geometry)%>%
  filter(!country=="Antarctica")

freedom12 <- freedom1%>%
  left_join(world12,by="country")
