library(tidyverse)
rent<-readRDS("rent.rds")
rent
max<-rent%>%
  mutate(city=str_to_title(city))%>%
  filter(year>2000)%>%
  group_by(year,city)%>%
  mutate(avg=median(price))%>%
  ungroup()%>%
  count(year,city,avg) %>%
    arrange(year,-avg) %>%
  slice_max(order_by=c(avg),n=10) %>%
  count(city) %>%
  select(-n) %>%
  unlist()

max
avg_df<- rent%>%
  mutate(city=str_to_title(city))%>%
  filter(year>2000)%>%
  group_by(year,city)%>%
  mutate(avg=median(price))%>%
  ungroup() %>%
  count(year,city,avg) %>%
  mutate(pct=round(n/sum(n)*100,2)) %>%
  arrange(year,-avg)
avg_group<-avg_df%>%
  filter(!city%in%max) %>%
  group_by(year)%>%
  summarise(avg_group=median(avg)) %>%
  ungroup()
summary(avg_df$avg)

text prices

text_max<-rent%>%
  mutate(city=str_to_title(city))%>%
  filter(year>2000)%>%
  group_by(year,city)%>%
  mutate(avg=median(price))%>%
  ungroup()%>%
  count(year,city,avg) %>%
    arrange(year,-avg) %>%
  slice_max(order_by=c(avg),n=5) %>%
  mutate(avg_comma=formatC(avg, format="d", big.mark=","))%>%
  mutate(avg_text=paste0("$",avg_comma," ",city))
